\begin{problem}
  Some calculus is
  \[
    \frac{d}{dx} \left( 2x^2 \right) = \answer{4x}.
  \]

  You can put \verb|\answer| in many places
    like \(\sqrt{\answer{9}} = 3\)
    or \[
      \frac{\answer{1}}{2} = 0.5.
    \]
\end{problem}


\begin{problem}
  A sage demo is
  \[
    \frac{d}{dx} \left( \sage{x*x + x*x} \right) =
      \answer{\sage{derivative(2*x^2,x)}}.
  \]
\end{problem}
