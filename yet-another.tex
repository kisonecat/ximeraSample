  x = 12;
  a = Math.round(Math.random()*10)
  b = Math.round(Math.random()*10)

Note $\js{x} = 12$ and $\js{3*3} = 9$.

\begin{problem}
  Consider $a = \js{a}$ and $b = \js{b}$.

  And then $\js{a} + \answer{\js{b-a}} = \js{b}$.
\end{problem}
