      return [x,1,-1][randint(0,2)]
    else:
      possible = [
        lambda: random_expression(depth-1) + random_expression(depth-1),
        lambda: random_expression(depth-1) - random_expression(depth-1),
        lambda: random_expression(depth-1) * random_expression(depth-1),
        lambda: random_expression(depth-1) / random_expression(depth-1),

        lambda: cos(random_expression(depth-1)),
        lambda: exp(random_expression(depth-1))
      ]
      result = possible[randint(0,len(possible)-1)]()
      if result.is_zero():
        result = 1
      return result
